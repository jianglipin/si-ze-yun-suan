int calculateIntegerResult(int num1, int num2, char signal)
{
    int t;
    switch (signal)
    {
    case '+':
        t = num1 + num2;
        break;
    case'-':
        t = num1 - num2;
        break;
    case'*':
        t = num1*num2;
        break;
    case'/':
        t = num1 / num2;
        break;
    default:
        t = 0;
        printf("运算符不符合规则\n");
        break;
    }
    return t;
}
